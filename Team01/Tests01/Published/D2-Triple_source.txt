procedure aaaProc {
    aaa = 1;
    ccc = bbb;
    call bbbProc;
    call cccProc;
    if ((aaa == bbb) && (!(aaa == ccc))) then {
        ddd = eee;
        print eee;
        call cccProc;
        while (fff != ggg) {
            hhh = (1 * (ggg / (ccc % ddd) + eee) - 2) + hhh / aaa;
            read ddd;
            call cccProc;
        }
        print hhh;
        aaa = 2;
    } else {
        read eee;
        while ((eee == eee) || (eee == eee)) {
            call bbbProc;
            if (bbb != ccc) then {
                aaa = ccc;
                bbb = ddd;
                call cccProc;
                read eee;
            } else {
                while (hhh != aaa) {
                    read aaa;
                    aaa = aaa + bbb - ccc * ddd / eee % fff * (ggg + hhh);
                    print aaa;
                }
                read aaa;
                print bbb;
                call cccProc;
            }
            read aaa;
            fff = fff / fff;
            ggg = hhh % hhh;
        }
        eee = fff;
        call bbbProc;
    }
    bbb = ccc / ((ddd + eee) * (fff % ggg - hhh));
    read aaa;
    print eee;
    call bbbProc;
    while (ddd == eee) {
        ccc = fff;
        if ((eee >= aaa) && (bbb < ddd)) then {
            hhh = ggg;
            print ggg;
        } else {
            eee = aaa * hhh;
            read fff;
        }
        call bbbProc;
    }
    read e;
    call cccProc;
}

procedure bbbProc {
    call dddProc;
    aaa = bbb;
    if (aaa == bbb) then {
        bbb = fff + ccc - ggg;
        read aaa;
        call dddProc;
    } else {
        ddd = eee;
        read ggg;
        call cccProc;
    }
    if (hhh > ggg) then {
        print ggg;
    } else {
        read ggg;
    }
    while (aaa <= ggg) {
        read fff;
        print aaa;
        call cccProc;
        call dddProc;
        call cccProc;
    }
    read hhh;
}

procedure cccProc {
    aaa = bbb;
    call dddProc;
    while (bbb >= eee) {
        while (bbb <= eee) {
            if (aaa == ggg) then {
                if (aaa != ggg) then {
                    call dddProc;
                } else {
                    call dddProc;
                }
            } else {
                while (ddd <= hhh) {
                    fff = ggg;
                }
            }
        }
    }
    print aaa;
    print bbb;
    print ccc;
    read ddd;
    read eee;
    call dddProc;
}

procedure dddProc {
    aaa = bbb + ccc;
    read ddd;
    print eee;
}
